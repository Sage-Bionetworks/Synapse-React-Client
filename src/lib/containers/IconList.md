<h4>Example using theme color</h4>
{
  <IconList iconConfigs={
    {
      'drugCombinationScreen': { icon: 'rat' },
      'geneExpression': { icon: 'geneExpression' },
    }

  useTheme={true}
  iconNames={["drugCombinationScreen", "geneExpression"]}
  />
}
